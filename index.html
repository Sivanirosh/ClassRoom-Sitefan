        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        // Utilisation de la configuration dynamique de l'environnement (Règle 1 & Évitement d'erreur API Key)

        const firebaseConfig = typeof __firebase_config !== 'undefined' 

            ? JSON.parse(__firebase_config) 

            : {

                apiKey: "",

                authDomain: "classe-numerique-prof.firebaseapp.com",

                projectId: "classe-numerique-prof",

                storageBucket: "classe-numerique-prof.appspot.com",

                messagingSenderId: "123456789",

                appId: "1:123456:web:abcde"

            };


        const app = initializeApp(firebaseConfig);

        const auth = getAuth(app);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';


        window.db = db;

        window.appId = appId;


        // Authentification Firebase (Règle 3)

        async function initAuth() {

            try {

                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

                    await signInWithCustomToken(auth, __initial_auth_token);

                } else {

                    await signInAnonymously(auth);

                }

            } catch (error) {

                console.error("Erreur d'authentification Firebase:", error);

            }

        }

        initAuth();


        onAuthStateChanged(auth, (user) => {

            if (user) {

                window.currentUser = user;

                chargerResultats();

            }

        });


        // Lecture des scores en temps réel (Règle 2)

        function chargerResultats() {

            if (!window.currentUser) return;

            const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

            

            onSnapshot(colRef, (snapshot) => {

                const liste = [];

                snapshot.forEach((doc) => {

                    liste.push({ id: doc.id, ...doc.data() });

                });

                // Tri par date décroissante

                liste.sort((a, b) => b.timestamp - a.timestamp);

                window.scoresData = liste;

                window.afficherTableauScores(liste);

            }, (error) => {

                console.error("Erreur lors de la lecture des scores :", error);

            });

        }

    </script>

    <script>

        function seConnecter() {

            const input = document.getElementById('student-name');

            const nom = input.value.trim();

            if (nom.length < 3) {

                alert("Veuillez entrer votre prénom complet ainsi que la première lettre de votre nom.");

                return;

            }

            localStorage.setItem('nomEleve', nom);

            document.getElementById('display-name').innerText = nom;

            document.getElementById('login-screen').classList.add('hidden');

        }


        function changerUtilisateur() {

            localStorage.removeItem('nomEleve');

            document.getElementById('login-screen').classList.remove('hidden');

        }


        window.onload = function() {

            const nomMemorise = localStorage.getItem('nomEleve');

            if (nomMemorise) {

                document.getElementById('display-name').innerText = nomMemorise;

                document.getElementById('login-screen').classList.add('hidden');

            }

        }


        function ouvrirActivite(page) {

            const nom = localStorage.getItem('nomEleve') || "Anonyme";

            window.location.href = `${page}?eleve=${encodeURIComponent(nom)}`;

        }


        function ouvrirAdmin() {

            const code = prompt("Veuillez saisir votre code d'accès enseignant :");

            if (code === "Prof2026") {

                document.getElementById('admin-panel').classList.remove('hidden');

            } else if (code !== null) {

                alert("Accès refusé. Code erroné.");

            }

        }


        function fermerAdmin() {

            document.getElementById('admin-panel').classList.add('hidden');

        }


        // Affiche le tableau de suivi enseignant

        window.afficherTableauScores = function(liste) {

            const tbody = document.getElementById('scores-table-body');

            if (liste.length === 0) {

                tbody.innerHTML = `<tr><td colspan="5" class="py-8 text-center text-slate-400 italic">Aucune tentative enregistrée pour le moment.</td></tr>`;

                return;

            }


            tbody.innerHTML = listTemplate(liste);

        }


        function listTemplate(liste) {

            return liste.map((item, idx) => {

                const date = item.timestamp ? new Date(item.timestamp).toLocaleString('fr-FR') : 'Date inconnue';

                const total = item.corrects + item.erreurs;

                const ratio = total > 0 ? (item.corrects / total * 100).toFixed(0) : 0;

                

                let badgeColor = 'bg-red-100 text-red-700';

                if (ratio >= 80) badgeColor = 'bg-green-100 text-green-700';

                else if (ratio >= 50) badgeColor = 'bg-amber-100 text-amber-700';


                return `

                    <tr onclick="ouvrirDetails(${idx})" class="border-b border-slate-100 hover:bg-blue-50/50 cursor-pointer transition">

                        <td class="py-4 px-4 font-bold text-slate-800">${item.nomEleve}</td>

                        <td class="py-4 px-4 text-slate-600">${item.exercice}</td>

                        <td class="py-4 px-4 text-center">

                            <span class="px-3 py-1 rounded-full text-xs font-bold ${badgeColor}">

                                ${item.corrects} / ${total} (${ratio}%)

                            </span>

                        </td>

                        <td class="py-4 px-4 text-sm text-slate-500">${date}</td>

                        <td class="py-4 px-4 text-right text-blue-600 font-bold hover:underline text-sm">Consulter ➔</td>

                    </tr>

                `;

            }).join('');

        }


        // Ouvre le détail complet d'une session

        window.ouvrirDetails = function(index) {

            const item = window.scoresData[index];

            if (!item) return;


            document.getElementById('modal-student-title').innerText = `Tentative de ${item.nomEleve}`;

            const date = item.timestamp ? new Date(item.timestamp).toLocaleString('fr-FR') : 'Date inconnue';

            document.getElementById('modal-exercise-subtitle').innerText = `${item.exercice} — Réalisé le ${date}`;


            const container = document.getElementById('modal-questions-list');

            container.innerHTML = "";


            if (!item.details || item.details.length === 0) {

                container.innerHTML = `<p class="text-slate-500 italic text-center py-6">Aucun détail disponible pour cette session.</p>`;

            } else {

                item.details.forEach((q, i) => {

                    const block = document.createElement('div');

                    block.className = "p-5 rounded-2xl border border-slate-100 bg-slate-50 space-y-3 shadow-sm";


                    // En-tête de la question

                    const header = document.createElement('div');

                    header.className = "flex justify-between items-start";

                    header.innerHTML = `

                        <span class="font-bold text-slate-800 text-md font-serif">Question ${i + 1} :</span>

                        <span class="px-2 py-1 rounded-lg text-xs font-bold ${q.isCorrect ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'}">

                            ${q.isCorrect ? 'Réussi' : 'Échec final'}

                        </span>

                    `;

                    block.appendChild(header);


                    // Intitulé de la question

                    const prompt = document.createElement('p');

                    prompt.className = "text-slate-700 font-semibold text-lg bg-white p-3 rounded-xl border border-slate-100";

                    prompt.innerText = q.questionText;

                    block.appendChild(prompt);


                    // Bonne réponse attendue

                    const ans = document.createElement('p');

                    ans.className = "text-sm text-slate-500";

                    ans.innerHTML = `Réponse attendue : <strong class="text-slate-800">${q.correctAnswer}</strong>`;

                    block.appendChild(ans);


                    // Journal des essais de l'élève

                    const attemptsWrapper = document.createElement('div');

                    attemptsWrapper.className = "space-y-1.5 pt-2 border-t border-slate-200/60";

                    

                    if (q.attempts && q.attempts.length > 0) {

                        q.attempts.forEach((att, attIdx) => {

                            const isLastAndWin = (attIdx === q.attempts.length - 1) && q.isCorrect;

                            const attDiv = document.createElement('div');

                            attDiv.className = "flex items-center gap-2 text-sm";

                            attDiv.innerHTML = `

                                <span class="text-xs text-slate-400">Essai ${attIdx + 1} :</span>

                                <span class="font-semibold text-slate-800">${att}</span>

                                <span>${isLastAndWin ? '✅' : '❌'}</span>

                            `;

                            attemptsWrapper.appendChild(attDiv);

                        });

                    } else {

                        attemptsWrapper.innerHTML = `<span class="text-xs text-slate-400 italic">Aucun essai enregistré.</span>`;

                    }

                    block.appendChild(attemptsWrapper);


                    container.appendChild(block);

                });

            }


            document.getElementById('detail-modal').classList.remove('hidden');

        }


        function fermerDetailModal() {

            document.getElementById('detail-modal').classList.add('hidden');

        }
