    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        const firebaseConfig = {

            apiKey: "",

            authDomain: "classe-numerique-prof.firebaseapp.com",

            projectId: "classe-numerique-prof",

            storageBucket: "classe-numerique-prof.appspot.com",

            messagingSenderId: "123456789",

            appId: "1:123456:web:abcde"

        };


        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';


        const urlParams = new URLSearchParams(window.location.search);

        const nomEleve = urlParams.get('eleve') || "Élève Mystère";

        window.nomEleve = nomEleve;


        // Fonction adaptative : Analyse l'historique de l'élève pour ajuster les types de questions

        async function analyserHistoriqueEtPreparer() {

            try {

                // Étape 1 : Récupérer TOUS les scores pour filtrer en mémoire locale (Règle 2)

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                const querySnapshot = await getDocs(colRef);

                const historique = [];

                

                querySnapshot.forEach(doc => {

                    const data = doc.data();

                    if (data.nomEleve === nomEleve && data.exercice === "Calcul Décimal") {

                        historique.push(data);

                    }

                });


                // Étape 2 : Analyser l'historique des questions problématiques

                // On va générer une liste de questions prioritaires (additions/soustractions ciblées)

                const erreursAdditions = historique.reduce((acc, current) => {

                    if (current.details) {

                        current.details.forEach(q => {

                            if (!q.isCorrect && q.questionText.includes('+')) acc++;

                        });

                    }

                    return acc;

                }, 0);


                const erreursSoustractions = historique.reduce((acc, current) => {

                    if (current.details) {

                        current.details.forEach(q => {

                            if (!q.isCorrect && q.questionText.includes('-')) acc++;

                        });

                    }

                    return acc;

                }, 0);


                window.generationDecimauxProfile = {

                    erreursAdditions,

                    erreursSoustractions

                };


                document.getElementById('loading-overlay').classList.add('hidden');

                nouveauCalcul();


            } catch (e) {

                console.error("Erreur d'analyse adaptative. Démarrage classique.", e);

                window.generationDecimauxProfile = { erreursAdditions: 0, erreursSoustractions: 0 };

                document.getElementById('loading-overlay').classList.add('hidden');

                nouveauCalcul();

            }

        }


        // Sauvegarder la tentative détaillée de l'exercice

        async function enregistrerSession(corrects, erreurs, detailQuestions) {

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                await addDoc(colRef, {

                    nomEleve: nomEleve,

                    exercice: "Calcul Décimal",

                    corrects: corrects,

                    erreurs: erreurs,

                    timestamp: Date.now(),

                    details: detailQuestions // Contient le log complet par question

                });

                console.log("Session complète enregistrée.");

            } catch (e) {

                console.error("Échec de la sauvegarde de session :", e);

            }

        }


        window.analyserHistoriqueEtPreparer = analyserHistoriqueEtPreparer;

        window.enregistrerSession = enregistrerSession;

    </script>

                   class="w-full max-w-xs px-4 py-4 border-2 border-slate-200 rounded-2xl text-center text-3xl font-bold focus:border-blue-500 focus:outline-none mb-4 shadow-sm">

            <br>

            <button id="validate-btn" onclick="verifier()" class="w-full bg-blue-600 text-white font-bold py-4 rounded-2xl hover:bg-blue-700 transition active:scale-95 shadow-md">Valider ma réponse</button>

            <button id="next-btn" onclick="prochaineQuestion()" class="w-full bg-emerald-500 text-white font-bold py-4 rounded-2xl hover:bg-emerald-600 transition hidden shadow-md">Continuer ➔</button>

            <button onclick="window.location.href='index.html'" class="w-full bg-blue-600 text-white font-bold py-4 rounded-2xl hover:bg-blue-700 transition shadow-md">Retourner au portail</button>

    <script>

        document.getElementById('student-tag').innerText = window.nomEleve;


        let solution;

        let corrects = 0;

        let erreurs = 0;

        let totalQuestions = 0;

        let essais = 0;

        

        // Stocke la trace d'apprentissage de cette session

        let detailsSession = [];

        let essaisActuelsQuestion = [];


        function nouveauCalcul() {

            essais = 0;

            essaisActuelsQuestion = [];

            document.getElementById('feedback').innerText = "";

            document.getElementById('reponse').value = "";

            document.getElementById('reponse').disabled = false;

            document.getElementById('validate-btn').classList.remove('hidden');

            document.getElementById('next-btn').classList.add('hidden');

            document.getElementById('badge-question').innerText = `Question ${totalQuestions + 1} / 10`;


            // Moteur adaptatif d'orientation de la question

            let op = '+';

            const prof = window.generationDecimauxProfile || { erreursAdditions: 0, erreursSoustractions: 0 };

            

            // Si l'élève a fait plus d'erreurs en soustractions par le passé, on en génère à 75% du temps, sinon 50%

            if (prof.erreursSoustractions > prof.erreursAdditions) {

                op = Math.random() < 0.75 ? '-' : '+';

            } else if (prof.erreursAdditions > prof.erreursSoustractions) {

                op = Math.random() < 0.75 ? '+' : '-';

            } else {

                op = Math.random() > 0.5 ? '+' : '-';

            }


            let n1 = (Math.random() * 19 + 1).toFixed(1);

            let n2 = (Math.random() * 9 + 1).toFixed(1);

            

            if (op === '-' && parseFloat(n1) < parseFloat(n2)) {

                [n1, n2] = [n2, n1];

            }


            solution = op === '+' ? parseFloat(n1) + parseFloat(n2) : parseFloat(n1) - parseFloat(n2);

            solution = parseFloat(solution.toFixed(1));


            document.getElementById('question').innerText = `${n1} ${op} ${n2}`;

            document.getElementById('reponse').focus();

        }


        function verifier() {

            const input = document.getElementById('reponse');

            const repEleve = parseFloat(input.value);

            const feedback = document.getElementById('feedback');


            if (isNaN(repEleve)) return;


            // Enregistrer l'essai de l'élève

            essaisActuelsQuestion.push(repEleve.toString());


            if (repEleve === solution) {

                feedback.innerText = "✅ Excellent ! Bonne réponse.";

                feedback.className = "mt-4 font-bold text-md text-green-600";

                corrects++;

                document.getElementById('score-c').innerText = corrects;

                

                // Enregistrement des détails de cette question

                detailsSession.push({

                    questionText: document.getElementById('question').innerText,

                    correctAnswer: solution.toString(),

                    attempts: [...essaisActuelsQuestion],

                    isCorrect: true

                });


                finaliserEtape();

            } else {

                essais++;

                if (essais < 2) {

                    feedback.innerText = "❌ Ce n'est pas tout à fait ça. Essaie encore !";

                    feedback.className = "mt-4 font-bold text-md text-orange-500";

                    input.value = "";

                    input.focus();

                } else {

                    feedback.innerText = `❌ Oups ! La bonne réponse était ${solution}.`;

                    feedback.className = "mt-4 font-bold text-md text-red-500";

                    erreurs++;

                    document.getElementById('score-w').innerText = erreurs;

                    

                    detailsSession.push({

                        questionText: document.getElementById('question').innerText,

                        correctAnswer: solution.toString(),

                        attempts: [...essaisActuelsQuestion],

                        isCorrect: false

                    });


                    finaliserEtape();

                }

            }

        }


        function finaliserEtape() {

            totalQuestions++;

            document.getElementById('reponse').disabled = true;

            document.getElementById('validate-btn').classList.add('hidden');

            document.getElementById('next-btn').classList.remove('hidden');

        }


        function prochaineQuestion() {

            if (totalQuestions >= 10) {

                document.getElementById('game-zone').classList.add('hidden');

                document.getElementById('finish-zone').classList.remove('hidden');

                document.getElementById('final-score').innerText = `${corrects} / 10`;

                

                // Envoi des détails structurés à Firestore

                window.enregistrerSession(corrects, erreurs, detailsSession);

            } else {

                nouveauCalcul();

            }

        }


        document.getElementById('reponse').addEventListener('keypress', function (e) {

            if (e.key === 'Enter') {

                if (!document.getElementById('validate-btn').classList.contains('hidden')) {

                    verifier();

                } else {

                    prochaineQuestion();

                }

            }

        });


        // Lancement après chargement asynchrone adaptatif

        window.analyserHistoriqueEtPreparer();

    </script>
