    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        const firebaseConfig = {

            apiKey: "",

            authDomain: "classe-numerique-prof.firebaseapp.com",

            projectId: "classe-numerique-prof",

            storageBucket: "classe-numerique-prof.appspot.com",

            messagingSenderId: "123456789",

            appId: "1:123456:web:abcde"

        };


        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';


        const urlParams = new URLSearchParams(window.location.search);

        const nomEleve = urlParams.get('eleve') || "Élève Mystère";

        window.nomEleve = nomEleve;


        // Récupérer l'historique et préparer la liste des figures cibles (Apprentissage Adaptatif)

        async function analyserHistoriqueGeometrie() {

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                const snapshot = await getDocs(colRef);

                const historique = [];


                snapshot.forEach(doc => {

                    const data = doc.data();

                    if (data.nomEleve === nomEleve && data.exercice === "Défi Géométrie") {

                        historique.push(data);

                    }

                });


                // Compter les échecs par type de figure pour donner une priorité

                const erreursFormes = {};

                historique.forEach(session => {

                    if (session.details) {

                        session.details.forEach(q => {

                            if (!q.isCorrect) {

                                erreursFormes[q.correctAnswer] = (erreursFormes[q.correctAnswer] || 0) + 1;

                            }

                        });

                    }

                });


                window.erreursFormesHistorique = erreursFormes;

                document.getElementById('loading-overlay').classList.add('hidden');

                determinerOrdreQuestions();


            } catch (e) {

                console.error("Erreur d'analyse. Mode normal.", e);

                window.erreursFormesHistorique = {};

                document.getElementById('loading-overlay').classList.add('hidden');

                determinerOrdreQuestions();

            }

        }


        async function enregistrerSession(corrects, erreurs, detailQuestions) {

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                await addDoc(colRef, {

                    nomEleve: nomEleve,

                    exercice: "Défi Géométrie",

                    corrects: corrects,

                    erreurs: erreurs,

                    timestamp: Date.now(),

                    details: detailQuestions

                });

                console.log("Session de géométrie sauvegardée.");

            } catch (e) {

                console.error("Échec de sauvegarde géométrie :", e);

            }

        }


        window.analyserHistoriqueGeometrie = analyserHistoriqueGeometrie;

        window.enregistrerSession = enregistrerSession;

    <script>

        document.getElementById('student-tag').innerText = window.nomEleve;


        // Base de données des questions

        const figuresBase = [

        let questionsDeLaSession = [];

        let indexForme = 0;

        let corrects = 0;

        let erreurs = 0;

        let essais = 0;


        let detailsSession = [];

        let essaisActuelsQuestion = [];


        // Algorithme adaptatif : Trier les questions pour placer celles qui ont causé le plus d'erreurs en tête de file

        function determinerOrdreQuestions() {

            const erreursHistorique = window.erreursFormesHistorique || {};

            

            // On calcule un score de priorité pour chaque figure de la base

            const poolTrie = [...figuresBase].map(fig => {

                const nbErreurs = erreursHistorique[fig.name] || 0;

                return { ...fig, priorite: nbErreurs + Math.random() * 0.1 }; // Petit aléatoire pour rompre les égalités

            });


            // On trie par priorité décroissante (les plus problématiques d'abord)

            poolTrie.sort((a, b) => b.priorite - a.priorite);


            // On retient 5 questions pour la session

            questionsDeLaSession = poolTrie.slice(0, 5);

            chargerForme();

        }


        function chargerForme() {

            essais = 0;

            essaisActuelsQuestion = [];

            document.getElementById('feedback').innerText = "";

            document.getElementById('next-btn').classList.add('hidden');

            document.getElementById('badge-question').innerText = `Forme ${indexForme + 1} / ${questionsDeLaSession.length}`;


            const f = questionsDeLaSession[indexForme];

            document.getElementById('svg-polygon').setAttribute('points', f.points);

            document.getElementById('question-text').innerText = f.text;


            const grid = document.getElementById('choices-grid');

            grid.innerHTML = "";


            f.options.forEach((opt) => {

                const btn = document.createElement('button');

                btn.className = "w-full bg-slate-50 border border-slate-200 text-slate-800 font-semibold py-3.5 px-4 rounded-xl hover:bg-blue-50 hover:border-blue-300 transition text-left";

                btn.innerText = opt;

                btn.onclick = () => verifier(opt, btn);

                grid.appendChild(btn);

            });

        }


        function verifier(nomChoisi, btnElement) {

            const f = questionsDeLaSession[indexForme];

            const feedback = document.getElementById('feedback');


            essaisActuelsQuestion.push(nomChoisi);


            const buttons = document.querySelectorAll('#choices-grid button');


            if (nomChoisi === f.correct) {

                feedback.innerText = f.desc;

                feedback.className = "font-bold text-md text-green-600 mb-4";

                btnElement.className = "w-full bg-green-100 border-2 border-green-500 text-green-800 font-bold py-3.5 px-4 rounded-xl text-left";

                corrects++;

                document.getElementById('score-c').innerText = corrects;

                

                detailsSession.push({

                    questionText: f.text,

                    correctAnswer: f.correct,

                    attempts: [...essaisActuelsQuestion],

                    isCorrect: true

                });


                terminerForme(buttons);

            } else {

                essais++;

                if (essais < 2) {

                    feedback.innerText = "❌ Ce n'est pas tout à fait ça ! Réobserve bien la forme.";

                    feedback.className = "font-bold text-md text-orange-500 mb-4";

                    btnElement.className = "w-full bg-orange-100 border border-orange-300 text-orange-700 font-semibold py-3.5 px-4 rounded-xl text-left";

                } else {

                    feedback.innerText = `❌ Oups ! La bonne réponse était : ${f.correct}.`;

                    feedback.className = "font-bold text-md text-red-500 mb-4";

                    btnElement.className = "w-full bg-red-100 border-2 border-red-500 text-red-800 font-bold py-3.5 px-4 rounded-xl text-left";

                    erreurs++;

                    document.getElementById('score-w').innerText = erreurs;


                    detailsSession.push({

                        questionText: f.text,

                        correctAnswer: f.correct,

                        attempts: [...essaisActuelsQuestion],

                        isCorrect: false

                    });


                    terminerForme(buttons);

                }

            }

        }


        function terminerForme(buttons) {

            buttons.forEach(btn => btn.disabled = true);

            document.getElementById('next-btn').classList.remove('hidden');

        }


        function prochaineForme() {

            indexForme++;

            if (indexForme >= questionsDeLaSession.length) {

                document.getElementById('game-zone').classList.add('hidden');

                document.getElementById('finish-zone').classList.remove('hidden');

                document.getElementById('final-score').innerText = `${corrects} / ${questionsDeLaSession.length}`;

                

                window.enregistrerSession(corrects, erreurs, detailsSession);

            } else {

                chargerForme();

            }

        }
