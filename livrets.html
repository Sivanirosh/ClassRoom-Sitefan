        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';


        const urlParams = new URLSearchParams(window.location.search);

        const nomEleve = urlParams.get('eleve') || "Élève Mystère";

        // Analyse l'historique de l'élève pour les tables afin de trier par ordre de difficulté d'apprentissage

        async function analyserHistoriqueLivrets() {

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                const querySnapshot = await getDocs(colRef);

                const historique = [];

                

                querySnapshot.forEach(doc => {

                    const data = doc.data();

                    if (data.nomEleve === nomEleve && data.exercice === "Les Livrets") {

                        historique.push(data);

                    }

                });


                // Compiler les échecs par couple de multiplication (ex: "7 x 8")

                const erreursMultiplications = {};

                historique.forEach(session => {

                    if (session.details) {

                        session.details.forEach(q => {

                            if (!q.isCorrect) {

                                // Nettoyage de la clé (ex: "7 x 8" ou "8 x 7")

                                const parts = q.questionText.split('x').map(p => parseInt(p.trim()));

                                if (parts.length === 2) {

                                    const low = Math.min(parts[0], parts[1]);

                                    const high = Math.max(parts[0], parts[1]);

                                    const cle = `${low} x ${high}`;

                                    erreursMultiplications[cle] = (erreursMultiplications[cle] || 0) + 1;

                                }

                            }

                        });

                    }

                });


                window.erreursMultiplicationsHistorique = erreursMultiplications;

                document.getElementById('loading-overlay').classList.add('hidden');

                genererQuestionsAdaptatives();


            } catch (e) {

                console.error("Échec d'analyse adaptative. Démarrage classique.", e);

                window.erreursMultiplicationsHistorique = {};

                document.getElementById('loading-overlay').classList.add('hidden');

                genererQuestionsAdaptatives();

            }

        }


        // Sauvegarde de la session de jeu dans Firestore

        async function enregistrerSession(corrects, erreurs, detailQuestions) {

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'scores');

                await addDoc(colRef, {

                    nomEleve: nomEleve,

                    exercice: "Les Livrets",

                    corrects: corrects,

                    erreurs: erreurs,

                    timestamp: Date.now(),

                    details: detailQuestions

                });

                console.log("Résultats de multiplication sauvegardés.");

            } catch (e) {

                console.error("Erreur d'écriture Firebase :", e);

            }

        }


        // Rendre les fonctions disponibles de manière globale en les attachant à l'objet window

        window.analyserHistoriqueLivrets = analyserHistoriqueLivrets;

        window.enregistrerSession = enregistrerSession;

            initialiserPythagoreTable();

            analyserHistoriqueLivrets();

        });

    <script>

        let solution;

        let corrects = 0;

        let erreurs = 0;

        let totalQuestions = 0;

        let essais = 0;


        let detailsSession = [];

        let essaisActuelsQuestion = [];

        let listeQuestionsCalculs = [];


        // Prépare et sélectionne 10 questions de multiplication adaptatives

        function genererQuestionsAdaptatives() {

            const erreursHistorique = window.erreursMultiplicationsHistorique || {};


            // Créer tous les couples possibles de 1x1 à 10x10

            let couples = [];

            for (let i = 1; i <= 10; i++) {

                for (let j = 1; j <= 10; j++) {

                    const low = Math.min(i, j);

                    const high = Math.max(i, j);

                    const cle = `${low} x ${high}`;

                    const nbErreurs = erreursHistorique[cle] || 0;


                    couples.push({

                        n1: i,

                        n2: j,

                        cle: cle,

                        // Plus il y a eu d'erreurs historiques, plus la priorité de sélection est élevée

                        priorite: nbErreurs + Math.random() * 0.2

                    });

                }

            }


            // Trier par priorité décroissante

            couples.sort((a, b) => b.priorite - a.priorite);


            // Retenir les 10 plus prioritaires pour cette session

            listeQuestionsCalculs = couples.slice(0, 10);

            chargerCalcul();

        }


        function chargerCalcul() {

            essais = 0;

            essaisActuelsQuestion = [];

            document.getElementById('feedback').innerText = "";

            document.getElementById('reponse').value = "";

            document.getElementById('reponse').disabled = false;

            document.getElementById('validate-btn').classList.remove('hidden');

            document.getElementById('next-btn').classList.add('hidden');

            document.getElementById('badge-question').innerText = `Calcul ${totalQuestions + 1} / 10`;


            const calc = listeQuestionsCalculs[totalQuestions];

            solution = calc.n1 * calc.n2;


            document.getElementById('question').innerText = `${calc.n1} × ${calc.n2}`;

            document.getElementById('reponse').focus();

        }


        function verifier() {

            const input = document.getElementById('reponse');

            const repEleve = parseInt(input.value);

            const feedback = document.getElementById('feedback');


            if (isNaN(repEleve)) return;


            essaisActuelsQuestion.push(repEleve.toString());


            if (repEleve === solution) {

                feedback.innerText = "✅ Bravo ! C'est tout à fait juste.";

                feedback.className = "mt-4 font-bold text-md text-green-600";

                corrects++;

                document.getElementById('score-c').innerText = corrects;


                detailsSession.push({

                    questionText: document.getElementById('question').innerText.replace('×', 'x'),

                    correctAnswer: solution.toString(),

                    attempts: [...essaisActuelsQuestion],

                    isCorrect: true

                });


                finaliserEtape();

            } else {

                essais++;

                if (essais < 2) {

                    feedback.innerText = "❌ Ce n'est pas le bon produit. Rappelle-toi de tes tables, ou consulte l'aide !";

                    feedback.className = "mt-4 font-bold text-md text-orange-500";

                    input.value = "";

                    input.focus();

                } else {

                    feedback.innerText = `❌ Oups ! La bonne réponse était ${solution}.`;

                    feedback.className = "mt-4 font-bold text-md text-red-500";

                    erreurs++;

                    document.getElementById('score-w').innerText = erreurs;


                    detailsSession.push({

                        questionText: document.getElementById('question').innerText.replace('×', 'x'),

                        correctAnswer: solution.toString(),

                        attempts: [...essaisActuelsQuestion],

                        isCorrect: false

                    });


                    finaliserEtape();

                }

            }

        }


        function finaliserEtape() {

            totalQuestions++;

            document.getElementById('reponse').disabled = true;

            document.getElementById('validate-btn').classList.add('hidden');

            document.getElementById('next-btn').classList.remove('hidden');

        }


        function prochainCalcul() {

            if (totalQuestions >= 10) {

                document.getElementById('game-zone').classList.add('hidden');

                document.getElementById('finish-zone').classList.remove('hidden');

                document.getElementById('final-score').innerText = `${corrects} / 10`;

                window.enregistrerSession(corrects, erreurs, detailsSession);

            } else {

                chargerCalcul();

            }

        }

        // Génère la table de Pythagore interactive au démarrage

        function initialiserPythagoreTable() {

            const grid = document.getElementById('pythagore-grid');

            grid.innerHTML = "";


            // Cellule d'angle vide

            const angle = document.createElement('div');

            angle.className = "bg-slate-200 text-slate-800 p-2 border border-slate-300 font-extrabold flex justify-center items-center rounded-tl-lg";

            angle.innerText = "×";

            grid.appendChild(angle);


            // Entête colonne de 1 à 10

            for (let c = 1; c <= 10; c++) {

                const headerCol = document.createElement('div');

                headerCol.className = "bg-slate-200 text-slate-700 p-2 border border-slate-300 font-extrabold flex justify-center items-center";

                headerCol.innerText = c;

                grid.appendChild(headerCol);

            }


            // Génération des lignes de la grille

            for (let r = 1; r <= 10; r++) {

                // Entête ligne

                const headerRow = document.createElement('div');

                headerRow.className = "bg-slate-200 text-slate-700 p-2 border border-slate-300 font-extrabold flex justify-center items-center";

                headerRow.innerText = r;

                grid.appendChild(headerRow);


                for (let col = 1; col <= 10; col++) {

                    const cellValue = r * col;

                    const cell = document.createElement('div');

                    cell.className = "bg-white p-2 border border-slate-200 hover:bg-violet-100 hover:text-violet-900 cursor-pointer font-bold flex justify-center items-center rounded transition-all";

                    cell.innerText = cellValue;

                    

                    cell.onclick = () => {

                        document.getElementById('selection-display').innerText = `${r} × ${col} = ${cellValue}`;

                    };


                    grid.appendChild(cell);

                }

            }

        }


        document.getElementById('reponse').addEventListener('keypress', function (e) {

            if (e.key === 'Enter') {

                if (!document.getElementById('validate-btn').classList.contains('hidden')) {

                    verifier();

                } else {

                    prochainCalcul();

                }

            }

        });

    </script>
